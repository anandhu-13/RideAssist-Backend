Schema : Customer

Attributes : [id, name, phone, email, vehicle_no, running_km]

1) api for listing all customer details

    url : http://127.0.0.1:8000/api/customers/
    
    method : get

    body : nill

    authorization : nill

2) api for adding new customer details

    url : http://127.0.0.1:8000/api/customers/

    method : post

    body : {
        name: "customer name",

        phone : "customer phone number",

        email : "customer email",

        vehicle_no : "customer vehicle_no",

        running_km : "customer running_km"
    }

    authorization : nill

3) api for fetching a customer detail

    url : http://127.0.0.1:8000/api/customers/{id}/

    method : get

    body : nill

    authorization : nill

4) api for updating customer detail

    url : http://127.0.0.1:8000/api/customers/{id}/

    method : put

    body : {
        name: "customer name",

        phone : "customer phone number",

        email : "customer email",

        vehicle_no : "customer vehicle_no",

        running_km : "customer running_km"
    }

    authorization : nill

5) api for deleting customer

    url : http://127.0.0.1:8000/api/customers/{id}/

    method : delete

    body : nill

    authorization : nill


6) api for fetching token

    url : http://127.0.0.1:8000/api/token/

    method : post

    body : {username,password}

    authorization : nill


7) api for fetching access token by passing refresh token

    url : http://127.0.0.1:8000/api/token/refresh/

    method : post

    body : {refresh_token}

    authorization : nill
